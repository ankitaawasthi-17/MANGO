1. TINY+CIFAR ON ALL ABLATIONS:

#!/bin/bash
# run_multiseed_multibuffer.sh

SEEDS=(42 43 44 45 46)
METHODS=("ours" "fixed_lambda" "no_reg" "er")
RESULTS_DIR="./results_ablation"

echo "======================================================="
echo "  Multi-Seed Multi-Buffer Ablation"
echo "======================================================="

# ── TinyImageNet: buffer 2000 and 4000 ───────────────────
TINY_BUFFERS=(2000 4000)

for BUFFER in "${TINY_BUFFERS[@]}"; do
    for METHOD in "${METHODS[@]}"; do
        for SEED in "${SEEDS[@]}"; do
            echo ""
            echo "  [TINY] Method=$METHOD | Buffer=$BUFFER | Seed=$SEED"
            python3 main_tiny.py \
                --method      $METHOD \
                --dataset     tiny \
                --seed        $SEED \
                --buffer_size $BUFFER \
                --results_dir $RESULTS_DIR/tiny_buf${BUFFER}
        done
    done
done

# ── CIFAR-100: buffer 1000, 2000, 4000 ───────────────────
CIFAR_BUFFERS=(1000 2000 4000)

for BUFFER in "${CIFAR_BUFFERS[@]}"; do
    for METHOD in "${METHODS[@]}"; do
        for SEED in "${SEEDS[@]}"; do
            echo ""
            echo "  [CIFAR] Method=$METHOD | Buffer=$BUFFER | Seed=$SEED"
            python3 main_tiny.py \
                --method      $METHOD \
                --dataset     cifar \
                --seed        $SEED \
                --buffer_size $BUFFER \
                --results_dir $RESULTS_DIR/cifar_buf${BUFFER}
        done
    done
done

echo ""
echo "======================================================="
echo "  Aggregating all results..."
echo "======================================================="

python3 - << 'EOF'
import numpy as np, os, glob

methods = ["ours", "fixed_lambda", "no_reg", "er"]

def load_metrics(path):
    m   = np.load(path)
    n   = m.shape[0]
    acc = np.mean(m[n-1, :n])
    bwt = np.mean([m[n-1, t] - m[t, t] for t in range(n-1)])
    aaa = np.mean([np.mean(m[t, :t+1])  for t in range(n)])
    wc  = np.mean([np.min(m[t,  :t+1])  for t in range(n)])
    return acc, bwt, aaa, wc

def fmt(vals):
    if not vals: return "    N/A   "
    return f"{np.mean(vals):5.2f}±{np.std(vals):.2f}"

configs = [
    ("tiny",  2000, "./results_ablation/tiny_buf2000"),
    ("tiny",  4000, "./results_ablation/tiny_buf4000"),
    ("cifar", 1000, "./results_ablation/cifar_buf1000"),
    ("cifar", 2000, "./results_ablation/cifar_buf2000"),
    ("cifar", 4000, "./results_ablation/cifar_buf4000"),
]

for dataset, buf, results_dir in configs:
    print(f"\n{'='*70}")
    print(f"  {dataset.upper()} | Buffer={buf}")
    print(f"{'='*70}")
    hdr = f"  {'Method':<14} {'CIL-ACC':>12} {'TIL-ACC':>12} {'CIL-BWT':>12} {'WC-Acc':>12} {'AAA':>12}"
    print(hdr)
    print("  " + "─" * (len(hdr) - 2))

    for method in methods:
        cil_accs, til_accs, bwts, wcs, aaas = [], [], [], [], []
        pattern = os.path.join(results_dir, dataset, method, "seed_*")
        for seed_dir in sorted(glob.glob(pattern)):
            cp = os.path.join(seed_dir, "acc_matrix_cil.npy")
            tp = os.path.join(seed_dir, "acc_matrix_til.npy")
            if not (os.path.exists(cp) and os.path.exists(tp)):
                continue
            ca, cb, caaa, cwc = load_metrics(cp)
            ta, *_            = load_metrics(tp)
            cil_accs.append(ca); til_accs.append(ta)
            bwts.append(cb);     aaas.append(caaa); wcs.append(cwc)

        print(f"  {method:<14} {fmt(cil_accs):>12} {fmt(til_accs):>12} "
              f"{fmt(bwts):>12} {fmt(wcs):>12} {fmt(aaas):>12}")

    if dataset == "tiny":
        print(f"\n  DER++ ref: CIL=11.61 | TIL=48.87 | BWT=-34.6 | WC=10.35 | AAA=17.85")

print("\nDone.")
EOF
